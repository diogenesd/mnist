# Criando arquitetura da rede
def base_model(num_pixels, num_classes):
	model = Sequential() # tipo keras senquencial
    
    # Adicionando uma camada densa (totalmente conectada) para entrada de dados, 
    # quantidade de neuronio = quantidade de pixels
    # inicializao os pesos da rede, modo normal = {'mean': 0.0, 'seed': None, 'stddev': 0.05}
    # Funo de ativao do neurnio ser Relu (https://en.wikipedia.org/wiki/Rectifier_(neural_networks))
	model.add(Dense(num_pixels, input_dim=num_pixels, kernel_initializer='normal', activation='relu'))
    
    # Adicionando uma camada densa (totalmente conectada) para sada de dados, 
    # inicializao os pesos da rede, modo normal = {'mean': 0.0, 'seed': None, 'stddev': 0.05}
    # Funo de ativao do neurnio ser Softmax (https://en.wikipedia.org/wiki/Softmax_function)
    # tipo da camada de sada  predio
	model.add(Dense(num_classes, kernel_initializer='normal', activation='softmax', name='preds'))
    
    # Copilando o modelo
    # Funo de apredizado ser cross-entropy (https://en.wikipedia.org/wiki/Cross_entropy)
    # Funo otimizadora descida gradiente estocstica (https://en.wikipedia.org/wiki/Stochastic_gradient_descent)
    # mtrica de reconhecimento ser preciso.
	model.compile(loss='categorical_crossentropy', optimizer='sgd', metrics=['accuracy'])
    
    # retorna o modelo compilado
	return model
